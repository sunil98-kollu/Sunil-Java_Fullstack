# Inner join - It returns the common data from 2 tables based on the common column-- syntax
select * from table1
inner join table2
on table1.commoncol = table2.commoncol;

select * from customers
inner join age
on customers.id=age.id;

-- using table name before common column name
select customer.id,name,age from customer
inner join age
on customer.ID = age.ID;
-- Writing joining using alais names
select * from customer as c
inner join age as a
on c.id = a.id;

on table1.commoncol = table2.commoncol
UNION
select * from table1

on table1.commoncol = table2.commoncol;
